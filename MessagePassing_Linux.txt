// sender.cpp
#include <sys/ipc.h>
#include <sys/msg.h>
#include <iostream>
#include <cstring>
using namespace std;
struct msg_buffer {
long msg_type;
char msg[100];
};
int main() {
key_t key = ftok("msgfile", 65);
int msgid = msgget(key, 0666 | IPC_CREAT);
msg_buffer message;
message.msg_type = 1;
cout << "Enter a message: ";
cin.getline(message.msg, 100);
msgsnd(msgid, &message, sizeof(message.msg), 0);
cout << "Message sent.\n";
return 0;
}




// receiver.cpp
#include <sys/ipc.h>
#include <sys/msg.h>
#include <iostream>
using namespace std;
struct msg_buffer {
long msg_type;
char msg[100];
};
int main() {
key_t key = ftok("msgfile", 65);
int msgid = msgget(key, 0666);
msg_buffer message;
msgrcv(msgid, &message, sizeof(message.msg), 1, 0);
cout << "Received Message: " << message.msg << endl;
msgctl(msgid, IPC_RMID, NULL); // delete queue
return 0;
}
